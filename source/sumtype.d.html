1 /++
2 [SumType] is a generic discriminated union implementation that uses
3 design-by-introspection to generate safe and efficient code. Its features
4 include:
5 
6 $(LIST
7     * [match|Pattern matching.]
8     * Support for self-referential types.

11     * A type-safe and memory-safe API compatible with DIP 1000 (`scope`).
12     * No dependency on runtime type information (`TypeInfo`).
13     * Compatibility with BetterC.

22 /// $(H3 Basic usage)
23 @safe unittest {
24     import std.math: isClose;
25 
26     struct Fahrenheit { double degrees; }
27     struct Celsius { double degrees; }
28     struct Kelvin { double degrees; }
29 
30     alias Temperature = SumType!(Fahrenheit, Celsius, Kelvin);
31 
32     // Construct from any of the member types.
33     Temperature t1 = Fahrenheit(98.6);
34     Temperature t2 = Celsius(100);
35     Temperature t3 = Kelvin(273);
36 
37     // Use pattern matching to access the value.
38     Fahrenheit toFahrenheit(Temperature t)
39     {
40         return Fahrenheit(
41             t.match!(
42                 (Fahrenheit f) => f.degrees,
43                 (Celsius c) => c.degrees * 9.0/5 + 32,
44                 (Kelvin k) => k.degrees * 9.0/5 - 459.4
45             )
46         );
47     }
48 
49     assert(toFahrenheit(t1).degrees.isClose(98.6));
50     assert(toFahrenheit(t2).degrees.isClose(212));
51     assert(toFahrenheit(t3).degrees.isClose(32));
52 
53     // Use ref to modify the value in place.
54     void freeze(ref Temperature t)
55     {
56         t.match!(
57             (ref Fahrenheit f) => f.degrees = 32,
58             (ref Celsius c) => c.degrees = 0,
59             (ref Kelvin k) => k.degrees = 273
60         );
61     }
62 
63     freeze(t1);
64     assert(toFahrenheit(t1).degrees.isClose(32));
65 
66     // Use a catch-all handler to give a default result.
67     bool isFahrenheit(Temperature t)
68     {
69         return t.match!(
70             (Fahrenheit f) => true,
71             _ => false
72         );
73     }
74 
75     assert(isFahrenheit(t1));
76     assert(!isFahrenheit(t2));
77     assert(!isFahrenheit(t3));

81 /** $(H3 Introspection-based matching)
82  *
83  * In the `length` and `horiz` functions below, the handlers for `match` do not
84  * specify the types of their arguments. Instead, matching is done based on how
85  * the argument is used in the body of the handler: any type with `x` and `y`
86  * properties will be matched by the `rect` handlers, and any type with `r` and
87  * `theta` properties will be matched by the `polar` handlers.
88  */
89 @safe unittest {
90     import std.math: isClose, cos, PI, sqrt;
91 
92     struct Rectangular { double x, y; }
93     struct Polar { double r, theta; }
94     alias Vector = SumType!(Rectangular, Polar);
95 
96     double length(Vector v)
97     {
98         return v.match!(
99             rect => sqrt(rect.x^^2 + rect.y^^2),
100             polar => polar.r
101         );
102     }
103 
104     double horiz(Vector v)
105     {
106         return v.match!(
107             rect => rect.x,
108             polar => polar.r * cos(polar.theta)
109         );
110     }
111 
112     Vector u = Rectangular(1, 1);
113     Vector v = Polar(1, PI/4);
114 
115     assert(length(u).isClose(sqrt(2.0)));
116     assert(length(v).isClose(1));
117     assert(horiz(u).isClose(1));
118     assert(horiz(v).isClose(sqrt(0.5)));
119 }

122 /** $(H3 Arithmetic expression evaluator)
123  *
124  * This example makes use of the special placeholder type `This` to define a
125  * [https://en.wikipedia.org/wiki/Recursive_data_type|recursive data type]: an
126  * [https://en.wikipedia.org/wiki/Abstract_syntax_tree|abstract syntax tree] for
127  * representing simple arithmetic expressions.

129 @system unittest {
130     import std.functional: partial;
131     import std.traits: EnumMembers;
132     import std.typecons: Tuple;
133 
134     enum Op : string
135     {
136         Plus  = "+",
137         Minus = "-",
138         Times = "*",
139         Div   = "/"
140     }
141 
142     // An expression is either
143     //  - a number,
144     //  - a variable, or
145     //  - a binary operation combining two sub-expressions.
146     alias Expr = SumType!(
147         double,
148         string,
149         Tuple!(Op, "op", This*, "lhs", This*, "rhs")
150     );
151 
152     // Shorthand for Tuple!(Op, "op", Expr*, "lhs", Expr*, "rhs"),
153     // the Tuple type above with Expr substituted for This.
154     alias BinOp = Expr.Types[2];
155 
156     // Factory function for number expressions
157     Expr* num(double value)
158     {
159         return new Expr(value);
160     }
161 
162     // Factory function for variable expressions
163     Expr* var(string name)
164     {
165         return new Expr(name);
166     }
167 
168     // Factory function for binary operation expressions
169     Expr* binOp(Op op, Expr* lhs, Expr* rhs)
170     {
171         return new Expr(BinOp(op, lhs, rhs));
172     }
173 
174     // Convenience wrappers for creating BinOp expressions
175     alias sum  = partial!(binOp, Op.Plus);
176     alias diff = partial!(binOp, Op.Minus);
177     alias prod = partial!(binOp, Op.Times);
178     alias quot = partial!(binOp, Op.Div);
179 
180     // Evaluate expr, looking up variables in env
181     double eval(Expr expr, double[string] env)
182     {
183         return expr.match!(
184             (double num) => num,
185             (string var) => env[var],
186             (BinOp bop) {
187                 double lhs = eval(*bop.lhs, env);
188                 double rhs = eval(*bop.rhs, env);
189                 final switch(bop.op) {
190                     static foreach(op; EnumMembers!Op) {
191                         case op:
192                             return mixin("lhs" ~ op ~ "rhs");
193                     }
194                 }
195             }
196         );
197     }
198 
199     // Return a "pretty-printed" representation of expr
200     string pprint(Expr expr)
201     {
202         import std.format;
203 
204         return expr.match!(
205             (double num) => "%g".format(num),
206             (string var) => var,
207             (BinOp bop) => "(%s %s %s)".format(
208                 pprint(*bop.lhs),
209                 cast(string) bop.op,
210                 pprint(*bop.rhs)
211             )
212         );
213     }
214 
215     Expr* myExpr = sum(var("a"), prod(num(2), var("b")));
216     double[string] myEnv = ["a":3, "b":4, "c":7];
217 
218     assert(eval(*myExpr, myEnv) == 11);
219     assert(pprint(*myExpr) == "(a + (2 * b))");

266 /**
267  * A [tagged union](https://en.wikipedia.org/wiki/Tagged_union) that can hold a
268  * single value from any of a specified set of types.
269  *
270  * The value in a `SumType` can be operated on using [match|pattern matching].
271  *
272  * To avoid ambiguity, duplicate types are not allowed (but see the
273  * [sumtype#basic-usage|"basic usage" example] for a workaround).
274  *
275  * The special type `This` can be used as a placeholder to create
276  * self-referential types, just like with `Algebraic`. See the
277  * [sumtype#arithmetic-expression-evaluator|"Arithmetic expression evaluator" example] for
278  * usage.
279  *
280  * A `SumType` is initialized by default to hold the `.init` value of its
281  * first member type, just like a regular union. The version identifier
282  * `SumTypeNoDefaultCtor` can be used to disable this behavior.
283  *
284  * See_Also: `std.variant.Algebraic`
285  */

501 			/**
502 			 * Assigns a value to a `SumType`.
503 			 *
504 			 * If any of the `SumType`'s members other than the one being assigned
505 			 * to contain pointers or references, it is possible for the assignment
506 			 * to cause memory corruption (see the
507 			 * ["Memory corruption" example](#memory-corruption) below for an
508 			 * illustration of how). Therefore, such assignments are considered
509 			 * `@system`.
510 			 *
511 			 * An individual assignment can be `@trusted` if the caller can
512 			 * guarantee that there are no outstanding references to any `SumType`
513 			 * members that contain pointers or references at the time the
514 			 * assignment occurs.
515 			 *
516 			 * Examples:
517 			 *
518 			 * $(H3 Memory corruption)
519 			 *
520 			 * This example shows how assignment to a `SumType` can be used to
521 			 * cause memory corruption in `@system` code. In `@safe` code, the
522 			 * assignment `s = 123` would not be allowed.
523 			 *
524 			 * ---
525 			 * SumType!(int*, int) s = new int;
526 			 * s.tryMatch!(
527 			 *     (ref int* p) {
528 			 *         s = 123; // overwrites `p`
529 			 *         return *p; // undefined behavior
530 			 *     }
531 			 * );
532 			 * ---
533 			 */

535 			{
536 				import core.lifetime: forward;
537 				import std.traits: hasIndirections, hasNested;
538 				import std.meta: AliasSeq, Or = templateOr;
539 
540 				alias OtherTypes =
541 					AliasSeq!(Types[0 .. tid], Types[tid + 1 .. $]);

543 					anySatisfy!(Or!(hasIndirections, hasNested), OtherTypes);

553 						Storage.memberName!T, ": __ctfe ? rhs : forward!rhs",

555 				} else {
556 					mixin("Storage newStorage = { ",
557 						Storage.memberName!T, ": forward!rhs",
558 					" };");
559 				}
560 
561 				storage = newStorage;
562 				tag = tid;
563 
564 				return this;
565 			}

1564 /**
1565  * Calls a type-appropriate function with the value held in a [SumType].
1566  *
1567  * For each possible type the [SumType] can hold, the given handlers are
1568  * checked, in order, to see whether they accept a single argument of that type.
1569  * The first one that does is chosen as the match for that type. (Note that the
1570  * first match may not always be the most exact match.
1571  * See [#avoiding-unintentional-matches|"Avoiding unintentional matches"] for
1572  * one common pitfall.)
1573  *
1574  * Every type must have a matching handler, and every handler must match at
1575  * least one type. This is enforced at compile time.
1576  *
1577  * Handlers may be functions, delegates, or objects with `opCall` overloads. If
1578  * a function with more than one overload is given as a handler, all of the
1579  * overloads are considered as potential matches.
1580  *
1581  * Templated handlers are also accepted, and will match any type for which they
1582  * can be [implicitly instantiated](https://dlang.org/glossary.html#ifti). See
1583  * [sumtype#introspection-based-matching|"Introspection-based matching"] for an
1584  * example of templated handler usage.
1585  *
1586  * If multiple [SumType]s are passed to `match`, their values are passed to the
1587  * handlers as separate arguments, and matching is done for each possible
1588  * combination of value types. See [#multiple-dispatch|"Multiple dispatch"] for
1589  * an example.
1590  *
1591  * Returns:
1592  *   The value returned from the handler that matches the currently-held type.
1593  *
1594  * See_Also: `std.variant.visit`
1595  */
1596 template match(handlers...)
1597 {
1598 	import std.typecons: Yes;
1599 
1600 	/**
1601 	 * The actual `match` function.
1602 	 *
1603 	 * Params:
1604 	 *   args = One or more [SumType] objects.
1605 	 */
1606 	auto ref match(SumTypes...)(auto ref SumTypes args)
1607 		if (allSatisfy!(isSumType, SumTypes) && args.length > 0)
1608 	{
1609 		return matchImpl!(Yes.exhaustive, handlers)(args);
1610 	}
1611 }

1613 /** $(H3 Avoiding unintentional matches)
1614  *
1615  * Sometimes, implicit conversions may cause a handler to match more types than
1616  * intended. The example below shows two solutions to this problem.
1617  */
1618 @safe unittest {
1619     alias Number = SumType!(double, int);
1620 
1621     Number x;
1622 
1623     // Problem: because int implicitly converts to double, the double
1624     // handler is used for both types, and the int handler never matches.
1625     assert(!__traits(compiles,
1626         x.match!(
1627             (double d) => "got double",
1628             (int n) => "got int"
1629         )
1630     ));
1631 
1632     // Solution 1: put the handler for the "more specialized" type (in this
1633     // case, int) before the handler for the type it converts to.
1634     assert(__traits(compiles,
1635         x.match!(
1636             (int n) => "got int",
1637             (double d) => "got double"
1638         )
1639     ));
1640 
1641     // Solution 2: use a template that only accepts the exact type it's
1642     // supposed to match, instead of any type that implicitly converts to it.
1643     alias exactly(T, alias fun) = function (arg) {
1644         static assert(is(typeof(arg) == T));
1645         return fun(arg);
1646     };
1647 
1648     // Now, even if we put the double handler first, it will only be used for
1649     // doubles, not ints.
1650     assert(__traits(compiles,
1651         x.match!(
1652             exactly!(double, d => "got double"),
1653             exactly!(int, n => "got int")
1654         )
1655     ));
1656 }

1815 /* The number that the dim-th argument's tag is multiplied by when
1816  * converting TagTuples to and from case indices ("caseIds").
1817  *
1818  * Named by analogy to the stride that the dim-th index into a
1819  * multidimensional static array is multiplied by to calculate the
1820  * offset of a specific element.
1821  */
1822 private size_t stride(size_t dim, lengths...)()
1823 {
1824 	import core.checkedint: mulu;
1825

1827 	bool overflow = false;
1828 
1829 	static foreach (i; 0 .. dim) {
1830 		result = mulu(result, lengths[i], overflow);

1834 	 * stride!(SumTypes.length), so as long as this overflow check

1838 	return result;

1841 private template matchImpl(Flag!"exhaustive" exhaustive, handlers...)
1842 {
1843 	auto ref matchImpl(SumTypes...)(auto ref SumTypes args)
1844 		if (allSatisfy!(isSumType, SumTypes) && args.length > 0)
1845 	{
1846 		alias stride(size_t i) = .stride!(i, Map!(typeCount, SumTypes));
1847 
1848 		alias TagTuple = .TagTuple!SumTypes;
1849 
1850 		/*
1851 		 * A list of arguments to be passed to a handler needed for the case
1852 		 * labeled with `caseId`.
1853 		 */
1854 		template handlerArgs(size_t caseId)
1855 		{
1856 			enum tags = TagTuple.fromCaseId(caseId);
1857 			enum argsFrom(size_t i: tags.length) = "";
1858 			enum argsFrom(size_t i) = "args[" ~ toCtString!i ~ "].get!(SumTypes[" ~ toCtString!i ~ "]" ~
1859 				".Types[" ~ toCtString!(tags[i]) ~ "])(), " ~ argsFrom!(i + 1);
1860 			enum handlerArgs = argsFrom!0;
1861 		}
1862 
1863 		/* An AliasSeq of the types of the member values in the argument list
1864 		 * returned by `handlerArgs!caseId`.
1865 		 *
1866 		 * Note that these are the actual (that is, qualified) types of the
1867 		 * member values, which may not be the same as the types listed in
1868 		 * the arguments' `.Types` properties.
1869 		 */
1870 		template valueTypes(size_t caseId)
1871 		{
1872 			enum tags = TagTuple.fromCaseId(caseId);
1873 
1874 			template getType(size_t i)
1875 			{
1876 				enum tid = tags[i];
1877 				alias T = SumTypes[i].Types[tid];
1878 				alias getType = typeof(args[i].get!T());
1879 			}
1880 
1881 			alias valueTypes = Map!(getType, Iota!(tags.length));
1882 		}
1883 
1884 		/* The total number of cases is
1885 		 *
1886 		 *   Π SumTypes[i].Types.length for 0 ≤ i < SumTypes.length
1887 		 *
1888 		 * Conveniently, this is equal to stride!(SumTypes.length), so we can
1889 		 * use that function to compute it.
1890 		 */
1891 		enum numCases = stride!(SumTypes.length);
1892 
1893 		/* Guaranteed to never be a valid handler index, since
1894 		 * handlers.length <= size_t.max.
1895 		 */
1896 		enum noMatch = size_t.max;
1897 
1898 		// An array that maps caseIds to handler indices ("hids").
1899 		enum matches = () {
1900 			size_t[numCases] matches;
1901 
1902 			// Workaround for dlang issue 19561
1903 			foreach (ref match; matches) {
1904 				match = noMatch;
1905 			}
1906 
1907 			static foreach (caseId; 0 .. numCases) {
1908 				static foreach (hid, handler; handlers) {
1909 					static if (canMatch!(handler, valueTypes!caseId)) {
1910 						if (matches[caseId] == noMatch) {
1911 							matches[caseId] = hid;
1912 						}
1913 					}
1914 				}
1915 			}
1916 
1917 			return matches;
1918 		}();
1919 
1920 		import std.algorithm.searching: canFind;
1921 
1922 		// Check for unreachable handlers
1923 		static foreach (hid, handler; handlers) {
1924 			static assert(matches[].canFind(hid),
1925 				"`handlers[" ~ toCtString!hid ~ "]` " ~
1926 				"of type `" ~ ( __traits(isTemplate, handler)
1927 					? "template"
1928 					: typeof(handler).stringof
1929 				) ~ "` " ~
1930 				"never matches"
1931 			);
1932 		}
1933 
1934 		// Workaround for dlang issue 19993
1935 		enum handlerName(size_t hid) = "handler" ~ toCtString!hid;
1936 
1937 		static foreach (size_t hid, handler; handlers) {
1938 			mixin("alias ", handlerName!hid, " = handler;");
1939 		}
1940 
1941 		immutable argsId = TagTuple(args).toCaseId;
1942 
1943 		final switch (argsId) {
1944 			static foreach (caseId; 0 .. numCases) {
1945 				case caseId:
1946 					static if (matches[caseId] != noMatch) {
1947 						return mixin(handlerName!(matches[caseId]), "(", handlerArgs!caseId, ")");
1948 					} else {
1949 						static if(exhaustive) {
1950 							static assert(false,
1951 								"No matching handler for types `" ~ valueTypes!caseId.stringof ~ "`");
1952 						} else {
1953 							throw new MatchException(
1954 								"No matching handler for types `" ~ valueTypes!caseId.stringof ~ "`");
1955 						}
1956 					}
1957 			}
1958 		}
1959 
1960 		assert(false, "unreachable");
1961 	}
1962 }

1966 /* A TagTuple represents a single possible set of tags that `args`
1967  * could have at runtime.
1968  *
1969  * Because D does not allow a struct to be the controlling expression
1970  * of a switch statement, we cannot dispatch on the TagTuple directly.
1971  * Instead, we must map each TagTuple to a unique integer and generate
1972  * a case label for each of those integers.
1973  *
1974  * This mapping is implemented in `fromCaseId` and `toCaseId`. It uses
1975  * the same technique that's used to map index tuples to memory offsets
1976  * in a multidimensional static array.
1977  *
1978  * For example, when `args` consists of two SumTypes with two member
1979  * types each, the TagTuples corresponding to each case label are:
1980  *
1981  *   case 0:  TagTuple([0, 0])
1982  *   case 1:  TagTuple([1, 0])
1983  *   case 2:  TagTuple([0, 1])
1984  *   case 3:  TagTuple([1, 1])
1985  *
1986  * When there is only one argument, the caseId is equal to that
1987  * argument's tag.
1988  */
1989 static struct TagTuple(SumTypes...)
1990 {
1991 	size_t[SumTypes.length] tags;
1992 	alias tags this;
1993 
1994 	alias stride(size_t i) = .stride!(i, Map!(typeCount, SumTypes));
1995 
1996 	invariant {
1997 		static foreach (i; 0 .. tags.length) {
1998 			assert(tags[i] < SumTypes[i].Types.length);
1999 		}
2000 	}
2001 
2002 	this(ref const(SumTypes) args)
2003 	{
2004 		static foreach (i; 0 .. tags.length) {
2005 			tags[i] = args[i].tag;
2006 		}
2007 	}
2008 
2009 	static TagTuple fromCaseId(size_t caseId)
2010 	{
2011 		TagTuple result;
2012 
2013 		// Most-significant to least-significant
2014 		static foreach_reverse (i; 0 .. result.length) {
2015 			result[i] = caseId / stride!i;
2016 			caseId %= stride!i;
2017 		}
2018 
2019 		return result;
2020 	}
2021 
2022 	size_t toCaseId()
2023 	{
2024 		size_t result;
2025 
2026 		static foreach (i; 0 .. tags.length) {
2027 			result += tags[i] * stride!i;
2028 		}
2029 
2030 		return result;
2031 	}
2032 }
